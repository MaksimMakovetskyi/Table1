CREATE TABLE IF NOT EXISTS 
	singers (id serial PRIMARY KEY, NAME VARCHAR(100) 
	NOT NULL unique
);

CREATE TABLE IF NOT EXISTS 
	albums (id serial PRIMARY KEY, NAME VARCHAR(100) 
	NOT NULL, singers_id INTEGER REFERENCES singers(id)
);

CREATE TABLE IF NOT EXISTS 
	track (id serial PRIMARY KEY, NAME VARCHAR(100) 
	NOT NULL, albums_id INTEGER REFERENCES albums(id)
);

	public.albums ADD COLUMN year VARCHAR (100)
;
